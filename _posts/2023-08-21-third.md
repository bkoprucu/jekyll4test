    public Flux<Count> listCounters(Integer page, Integer size) {
        final int pg = page == null || page < 1 ? 0 : page;
        final int sz = size == null || size < 1 ? DEFAULT_ITEMS_PER_PAGE

        return Flux.fromStream(() ->

                                     .skip((long) pg * sz)
